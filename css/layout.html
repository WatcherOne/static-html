左边定宽，右边自适应
float + margin
.left {
    float: left;
    width: 120px;
}
.right {
    margin-left: 120px;
}
float + calc
.left {
    float: left;
    width: 120px;
}
.right {
    float: left;
    width: calc(100% - 120px)
}
inline-block + calc
.parent-box {
    font-size: 0px
    // 注意行内元素之间有默认间隙
}
.left {
    display: inline-block;
    width: 120px;
}
.right {
    display: inline-block;
    width: calc(100% - 120px)
}
flex
.parent-box {
    display: flex;
    flex-wrap: nowrap;
}
.left {
    width: 120px;
}
.right {
    flex: 1;
}
